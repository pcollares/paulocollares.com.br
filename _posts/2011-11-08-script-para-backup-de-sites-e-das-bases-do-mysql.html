Esse é um script bem simples para fazer backup do seu site, e das bases de dados do mySQL

#! /bin/bash

################################
# Script de backup do site	#
# Paulo Collares                #
# www.paulocollares.com.br      #
# 04/04/2011		       	#
################################

##Backup do mysql
mysqldump -u root -pSENHA_DO_MYSQL NOME_DA_BASE_DE_DADOS > /backup/NOME_DO_ARQUIVO.sql

##Backup do diretorio /var/www
tar -cvf /backup/www.tar /var/www/*
