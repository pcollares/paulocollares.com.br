Script muito útil para fazer backup de servidores externos por SFTP

#!/bin/sh

cd /backup #abre o diretório para salvar o backup
rm * #remove o backup antigo

sshpass -p SENHA_DO_USUARIO_ROOT sftp root@IP_DO_SERVIDOR << ENDSHERE
cd /backup #diretório remoto
get *
ENDSHERE
